  --main_process_port=29500 \
  /mars_data/zuo/envs/unitree_groot/bin/lerobot-train \
  --policy.pretrained_path=/mars_data/zuo/Robot_Gaze/unitree_lerobot/lerobot/outputs/train/groot_handover/checkpoints/020000/pretrained_model \
  --dataset.repo_id=ZUO66/handover_two_drinks \
  --dataset.root=/mars_data/zuo/my_data/handover_two_drinks \
  --policy.type=groot \
  --policy.repo_id=ZUO66/groot_mask \
  --policy.push_to_hub=false \
  --output_dir=outputs/train/groot_mask \
  --steps=50000 --batch_size=32 --num_workers=8 --save_freq=5000 --save_checkpoint=true \
  --wandb.enable=true --wandb.project=zuo_handover --job_name=groot_full_align_2gpu \

  --policy.device=cuda --policy.use_amp=true --resume=false --dataset.video_backend=pyav \
